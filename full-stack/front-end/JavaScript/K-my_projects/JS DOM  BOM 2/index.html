NORMAL FUNCTION

Syntax :-

function function_name(Arguments) {
Operations
}
function_name(parameters)

example

function f2(a,b) {
return a*b
}
console.log(f2(5,6))

ARROW FUNCTION

Syntax :-

function function_name(Arguments) => {
Operations
}
function_name(parameters)

example

let f3=(a,b) => {
return a*b
}
console.log(f3(3,6));
